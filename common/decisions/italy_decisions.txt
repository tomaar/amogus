ITA_emperors_assassination_investigation = {
	ITA_emperors_assassination_investigation_ticker = {
		icon = generic_industry
		
		days_remove = 14

		remove_effect = {
			set_temp_variable = {
				var = ITA_temp_unrest_change
				value = 5
			}
			add_to_variable = {
				var = ITA_assassination_nationalist_unrest
				value = ITA_temp_unrest_change
				tooltip = assassination_nationalist_unrest_tooltip
			}
			if = {
				limit = {
					check_variable = {
						var = ITA_assassination_nationalist_unrest
						value = 100
						compare = greater_than_or_equals
					}
				}
				country_event = {
					id = italy_assassination.5
				}
			}
			else_if = {
				limit = {
					check_variable = {
						var = ITA_assassination_arab_unrest
						value = 100
						compare = greater_than_or_equals
					}
				}
				country_event = {
					id = italy_assassination.6
				}
			}
			else = {
				activate_decision = ITA_emperors_assassination_investigation_ticker
			}
		}
	}

	ITA_emperors_assassination_investigation_take_over = {
		icon = generic_industry
		
		fire_only_once = yes

		visible = {
			NOT = {
				has_country_flag = italy_leading_the_investigation
			}
		}

		complete_effect = {
			set_country_flag = italy_leading_the_investigation
			custom_effect_tooltip = ITA_take_over_the_investigation
			
			set_temp_variable = {
				var = ITA_temp_unrest_change
				value = -20
			}
			add_to_variable = {
				var = ITA_assassination_nationalist_unrest
				value = ITA_temp_unrest_change
				tooltip = assassination_nationalist_unrest_tooltip
			}
			set_temp_variable = {
				var = ITA_temp_unrest_change
				value = 20
			}
			add_to_variable = {
				var = ITA_assassination_arab_unrest
				value = ITA_temp_unrest_change
				tooltip = assassination_arab_unrest_tooltip
			}
		}
	}
}
